function copyCode(element) {
                    var codeBlock = element.querySelector('code');
                    var textArea = document.createElement('textarea');
                    textArea.value = codeBlock.textContent;
                    document.body.appendChild(textArea);
                    textArea.select();
                    document.execCommand('copy');
                    document.body.removeChild(textArea);
                    
                    // Toggle the "copied" class to change button appearance
                    var copyButton = element.querySelector('.copy-button');
                    var copiedText = element.querySelector('.code-copied');
                    copyButton.classList.add('copied'); // Add "copied" class
                    copyButton.innerHTML = '✔ Code Copied'; // Display a tick mark icon
                    copiedText.textContent = 'Code Copied'; // Display "Code Copied" text
                    
                    // Reset the button appearance after a few seconds
                    setTimeout(function() {
                        copyButton.classList.remove('copied'); // Remove "copied" class
                        copyButton.innerHTML = 'ðŸ“‹ Copy'; // Reset button content
                        copiedText.textContent = ''; // Clear the "Code Copied" text
                    }, 2000); // Change back after 2 seconds (2000 milliseconds)
                }
